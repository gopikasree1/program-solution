import React from 'react';

function TeamList() {
  const teams = ['India', 'Australia', 'England', 'South Africa'];

  return (
    <div>
      <h2>Teams</h2>
      <ul>
        {teams.map((team, index) => (
          <li key={index}>{team}</li>
        ))}
      </ul>
    </div>
  );
}
export default TeamList;
